6. Building the RPG Step-by-Step

Now that you know the basics, let's build the RPG step-by-step.

Step 1: Define the Player and Enemy Classes

class Player:
    def __init__(self, name):
        self.name = name
        self.hp = 100
        self.attack = 10
        self.defense = 5
        self.alive = True

    def take_damage(self, damage):
        self.hp -= damage
        if self.hp <= 0:
            self.hp = 0
            self.alive = False

    def attack_enemy(self, enemy):
        damage = self.attack - enemy.defense
        if damage > 0:
            enemy.take_damage(damage)
            print(f"{self.name} attacks {enemy.name} for {damage} damage.")
        else:
            print(f"{self.name}'s attack was too weak to harm {enemy.name}.")

class Enemy:
    def __init__(self, name, hp, attack, defense):
        self.name = name
        self.hp = hp
        self.attack = attack
        self.defense = defense
        self.alive = True

    def take_damage(self, damage):
        self.hp -= damage
        if self.hp <= 0:
            self.hp = 0
            self.alive = False

    def attack_player(self, player):
        damage = self.attack - player.defense
        if damage > 0:
            player.take_damage(damage)
            print(f"{self.name} attacks {player.name} for {damage} damage.")
        else:
            print(f"{self.name}'s attack was too weak to harm {player.name}.")

Step 2: Create Combat Function

def print_status(player, enemy):
    print(f"{player.name}: HP={player.hp}")
    print(f"{enemy.name}: HP={enemy.hp}")

def combat(player, enemy):
    print(f"A wild {enemy.name} appears!")
    while player.alive and enemy.alive:
        print_status(player, enemy)
        action = input("Do you want to (1) attack or (2) defend? ")
        if action == "1":
            player.attack_enemy(enemy)
        else:
            print(f"{player.name} is defending.")
            player.defense += 5  # Temporary defense boost for defending

        if enemy.alive:
            enemy.attack_player(player)
        if action == "2":
            player.defense -= 5  # Remove the temporary defense boost

    if player.alive:
        print(f"{player.name} has defeated {enemy.name}!")
    else:
        print(f"{player.name} has been defeated by {enemy.name}.")

Step 3: Create the Adventure Function

def start_adventure():
    print("Welcome to the adventure game!")
    player_name = input("Enter your character's name: ")
    player = Player(player_name)

    print(f"{player.name}, you find yourself in a dark forest.")
    print("A path splits into two directions.")
    choice = input("Do you want to go (1) left or (2) right? ")

    if choice == "1":
        print(f"{player.name} takes the left path and encounters a goblin!")
        goblin = Enemy("Goblin", 30, 5, 2)
        combat(player, goblin)
    else:
        print(f"{player.name} takes the right path and encounters a wolf!")
        wolf = Enemy("Wolf", 50, 7, 3)
        combat(player, wolf)

    if player.alive:
        print(f"{player.name} continues deeper into the forest and finds a treasure chest.")
        print("Congratulations, you've completed the adventure!")
    else:
        print("Game Over. Better luck next time!")

if __name__ == "__main__":
    start_adventure()
